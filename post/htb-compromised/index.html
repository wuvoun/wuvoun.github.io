5. The malware used HTTPS traffic with a self-signed certificate. What are the ports, from smallest to largest?

tls.handshake.type == 11 是专门筛选 TLS 握手里的 证书传输报文。

#!/usr/bin/env bash
# find_selfsigned_with_attrs.sh
# 用法: ./find_selfsigned_with_attrs.sh capture.pcap
set -euo pipefail

PCAP="${1:-}"
if [ -z "$PCAP" ] || [ ! -f "$PCAP" ]; then
  echo "Usage: $0 <capture.pcap>"
  exit 1
fi

WORKDIR=$(mktemp -d)
trap 'rm -rf "$WORKDIR"' EXIT

OUT="$WORKDIR/out.tsv"
> "$OUT"

# 导出每个 TLS Certificate 报文的 src/dst ports 与证书 hex（每行一条记录）
tshark -r "$PCAP" -Y "tls.handshake.type == 11" \
  -T fields -e ip.src -e ip.dst -e tcp.srcport -e tcp.dstport -e tls.handshake.certificate \
  | sed '/^\s*$/d' > "$WORKDIR/raw_lines.txt"

n=0

# helper: 从 openssl subject/issuer 字符串中提取指定 attr（处理多种分隔符）
extract_attr() {
  local line="$1" attr="$2" val
  # remove leading "subject=" / "issuer=" if present
  line=$(printf '%s' "$line" | sed -e 's/^[[:space:]]*subject=//I' -e 's/^[[:space:]]*issuer=//I')
  # normalize separators to newline: commas, slash, semicolon
  # also change occurrences of ", " to ","
  # then split into lines and find first matching attribute (case-insensitive)
  printf '%s' "$line" \
    | sed -E 's/,\s*/\n/g; s#/\s*#\n#g; s/;\s*/\n/g' \
    | while IFS= read -r part; do
        # trim
        part=$(printf '%s' "$part" | sed -E 's/^[[:space:]]+//; s/[[:space:]]+$//')
        # some entries might be like "CN = value" or "CN=value"
        # normalize spaces around '='
        norm=$(printf '%s' "$part" | sed -E 's/[[:space:]]*=[[:space:]]*/=/')
        # check start of token
        case "$norm" in
          "${attr}="*|${attr,,}=*|${attr^^}=*) val="${norm#*=}"; printf '%s' "$val"; return 0 ;;
        esac
      done
  return 1
}

while IFS=$'\t' read -r IPS IPD PS PD HEX; do
  [ -z "${HEX:-}" ] && n=$((n+1)) && continue

  # write meta
  echo -e "${IPS}\t${IPD}\t${PS}\t${PD}" > "$WORKDIR/meta_$n.txt"

  # hex -> der (robust: strip whitespace if needed)
  if ! printf "%s" "$HEX" | xxd -r -p > "$WORKDIR/cert_$n.der" 2>/dev/null; then
    if printf "%s" "$HEX" | tr -d ' \t\r\n' | xxd -r -p > "$WORKDIR/cert_$n.der" 2>/dev/null; then
      :
    else
      echo "warn: failed to convert hex on line $n" >&2
      n=$((n+1)); continue
    fi
  fi

  # get basic openssl outputs (try direct, else parse first sequence)
  if ! openssl x509 -in "$WORKDIR/cert_$n.der" -inform DER -noout -subject -issuer -startdate -enddate -serial > "$WORKDIR/cert_$n.info" 2>/dev/null; then
    ASNOUT=$(openssl asn1parse -inform DER -in "$WORKDIR/cert_$n.der" 2>/dev/null | head -n1 || true)
    if [[ -n "$ASNOUT" ]]; then
      L=$(echo "$ASNOUT" | sed -n 's/.*l=\s*\([0-9]*\).*/\1/p')
      if [[ -n "$L" && "$L" -gt 0 ]]; then
        dd if="$WORKDIR/cert_$n.der" of="$WORKDIR/cert_${n}_first.der" bs=1 count="$L" 2>/dev/null || true
        if ! openssl x509 -in "$WORKDIR/cert_${n}_first.der" -inform DER -noout -subject -issuer -startdate -enddate -serial > "$WORKDIR/cert_$n.info" 2>/dev/null; then
          echo "warn: openssl x509 still failed for record $n" >&2
          n=$((n+1)); continue
        fi
      else
        echo "warn: couldn't determine top-level length for record $n" >&2
        n=$((n+1)); continue
      fi
    else
      echo "warn: asn1parse empty for record $n" >&2
      n=$((n+1)); continue
    fi
  fi

  # read extracted lines
  SUBJECT=$(sed -n 's/^subject= *//Ip' "$WORKDIR/cert_$n.info" | tr -d '\r\n')
  ISSUER=$(sed -n 's/^issuer= *//Ip' "$WORKDIR/cert_$n.info" | tr -d '\r\n')
  START=$(sed -n 's/^notBefore= *//Ip' "$WORKDIR/cert_$n.info" | tr -d '\r\n' || true)
  END=$(sed -n 's/^notAfter= *//Ip' "$WORKDIR/cert_$n.info" | tr -d '\r\n' || true)
  SERIAL=$(sed -n 's/^serial= *//Ip' "$WORKDIR/cert_$n.info" | tr -d '\r\n' || true)

  # compare subject and issuer textually (exact match)
  if [[ -n "$SUBJECT" && -n "$ISSUER" && "$SUBJECT" == "$ISSUER" ]]; then
    # extract specific attributes
    CN=$(extract_attr "$SUBJECT" "CN" || true)
    O=$(extract_attr "$SUBJECT" "O" || true)
    OU=$(extract_attr "$SUBJECT" "OU" || true)
    L=$(extract_attr "$SUBJECT" "L" || true)
    ST=$(extract_attr "$SUBJECT" "ST" || true)
    C=$(extract_attr "$SUBJECT" "C" || true)

    # ensure tabs are escaped/removed in extracted values
    CN=$(printf '%s' "$CN" | tr '\t' ' ')
    O=$(printf '%s' "$O" | tr '\t' ' ')
    OU=$(printf '%s' "$OU" | tr '\t' ' ')
    L=$(printf '%s' "$L" | tr '\t' ' ')
    ST=$(printf '%s' "$ST" | tr '\t' ' ')
    C=$(printf '%s' "$C" | tr '\t' ' ')
    START=$(printf '%s' "$START" | tr '\t' ' ')
    END=$(printf '%s' "$END" | tr '\t' ' ')
    SERIAL=$(printf '%s' "$SERIAL" | tr '\t' ' ')

    # Output TSV: ip.src	ip.dst	srcport	dstport	serial	startdate	enddate	C	ST	L	O	OU	CN
    printf '%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\n' \
      "$IPS" "$IPD" "$PS" "$PD" "$SERIAL" "$START" "$END" "$C" "$ST" "$L" "$O" "$OU" "$CN" \
      >> "$OUT"
  fi

  n=$((n+1))
done < "$WORKDIR/raw_lines.txt"

if [ ! -s "$OUT" ]; then
  echo "没有发现自签名证书流量（subject==issuer）。"
  exit 0
fi

# 生成分离的端口/IP 文件（去重&升序）
SRC_PORTS_FILE="$WORKDIR/selfsigned_src_ports.txt"
DST_PORTS_FILE="$WORKDIR/selfsigned_dst_ports.txt"
SRC_IP_PORT="$WORKDIR/selfsigned_src_ip_port.txt"
DST_IP_PORT="$WORKDIR/selfsigned_dst_ip_port.txt"
CSV_OUT="$WORKDIR/selfsigned_detailed.tsv"

cut -f3 "$OUT" | sed '/^$/d' | sort -n -u > "$SRC_PORTS_FILE"
cut -f4 "$OUT" | sed '/^$/d' | sort -n -u > "$DST_PORTS_FILE"
awk -F'\t' '{print $1 ":" $3}' "$OUT" | sort -u > "$SRC_IP_PORT"
awk -F'\t' '{print $2 ":" $4}' "$OUT" | sort -u > "$DST_IP_PORT"

# copy detailed out to human-friendly TSV with header
printf 'ip.src\tip.dst\tsrcport\tdstport\tserial\tstartdate\tenddate\tC\tST\tL\tO\tOU\tCN\n' > "$CSV_OUT"
cat "$OUT" >> "$CSV_OUT"

# Print results
echo "检测到以下自签名证书连接（每行为 TSV，字段：ip.src ip.dst srcport dstport serial startdate enddate C ST L O OU CN）:"
cat "$CSV_OUT"
echo

echo "源端口（srcport，去重并升序，每行一个）:"
cat "$SRC_PORTS_FILE"
echo
echo "目标端口（dstport，去重并升序，每行一个）:"
cat "$DST_PORTS_FILE"
echo

echo "已生成 IOC 列表（去重）："
echo "源 IP:port -> $SRC_IP_PORT"
echo "目标 IP:port -> $DST_IP_PORT"
echo
echo "详细 TSV 文件 -> $CSV_OUT"
echo
# optional: copy to current directory (uncomment if desired)
# cp "$SRC_PORTS_FILE" ./selfsigned_src_ports.txt
# cp "$DST_PORTS_FILE" ./selfsigned_dst_ports.txt
# cp "$SRC_IP_PORT" ./selfsigned_src_ip_port.txt
# cp "$DST_IP_PORT" ./selfsigned_dst_ip_port.txt
# cp "$CSV_OUT" ./selfsigned_detailed.tsv

echo "完成：临时输出文件保存在 $WORKDIR（脚本结束时该目录会被清理）。"

这个脚本用来分析流量包所有的tls的证书流量，发现自签名证书，将这些流量的信息打印出来
